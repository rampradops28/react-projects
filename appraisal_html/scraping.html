Web Scraping to JSON

This HTML file demonstrates how to use Python to scrape data from a website and convert it to JSON format.

Instructions

  1. Replace the placeholder values in the code below with your target website details.
  2. Save this file as a Python file (e.g., "scraper.py").
  3. Run the Python file from your terminal using "python scraper.py".

Python Code

from bs4 import BeautifulSoup
import requests
import json

def scrape_website(url, target_selector, data_attributes):
    """
    Scrapes data from a website and returns it as a JSON object.

    Args:
        url (str): The URL of the website to scrape.
        target_selector (str): A CSS selector to target the elements containing the desired data.
        data_attributes (dict): A dictionary where keys are data field names and values are CSS selectors 
                                within the target element to extract the data.

    Returns:
        dict: A dictionary containing the scraped data in JSON format.
    """

    response = requests.get(url)
    response.raise_for_status()  # Raise an exception for HTTP errors
    soup = BeautifulSoup(response.content, 'html.parser')

    data = []
    for item in soup.select(target_selector):
        item_data = {}
        for field, selector in data_attributes.items():
            element = item.select_one(selector)
            if element:
                item_data[field] = element.text.strip()
            else:
                item_data[field] = None  # Or handle missing data as needed

        data.append(item_data)

    return json.dumps(data, indent=4)  # Convert to JSON string with indentation

# Example Usage
url = "https://example.com/data"  # Replace with the actual URL
target_selector = ".product-item"  # CSS selector to target the items
data_attributes = {
    "product_name": ".product-name",
    "price": ".price",
    "description": ".product-description",
    # Add more data fields and their CSS selectors as needed
}

json_data = scrape_website(url, target_selector, data_attributes)
print(json_data)
